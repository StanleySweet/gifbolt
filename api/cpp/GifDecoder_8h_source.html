16/// \enum DisposalMethod
17/// \brief Specifies how to dispose of a frame before rendering the next one.
18enum class DisposalMethod : uint8_t
19{
20 None = 0, ///< No disposal specified (leave pixels as-is)
21 DoNotDispose = 1, ///< Leave frame pixels in place
22 RestoreBackground = 2, ///< Clear frame area to background color
23 RestorePrevious = 3 ///< Restore to previous frame state
24};
25
26/// \struct GifFrame
27/// \brief Represents a single frame in a GIF image.
28/// Contains pixel data and timing information for rendering.
29struct GifFrame
30{
31 std::vector<uint32_t> pixels; ///< RGBA pixel data (32 bits per pixel)
32 uint32_t width; ///< Frame width in pixels
33 uint32_t height; ///< Frame height in pixels
34 uint32_t offsetX; ///< Frame horizontal offset within canvas
35 uint32_t offsetY; ///< Frame vertical offset within canvas
36 uint32_t delayMs; ///< Display duration in milliseconds
37 DisposalMethod disposal; ///< Frame disposal method
38 int32_t transparentIndex; ///< Index of transparent color (-1 if none)
39};

41/// \class GifDecoder
42/// \brief Decodes GIF images from files or URLs.
43///
44/// Provides functionality to load and decode GIF images, with support for
45/// frame iteration, looping detection, and pixel data access.
46class GifDecoder
47{
48 public:
49 /// \brief Définit le délai minimal appliqué à chaque frame GIF (en ms).
50 /// \param minDelayMs Délai minimal en millisecondes.
51 void SetMinFrameDelayMs(uint32_t minDelayMs);
52
53 /// \brief Obtient le délai minimal appliqué à chaque frame GIF (en ms).
54 /// \return Délai minimal en millisecondes.
55 uint32_t GetMinFrameDelayMs() const;
56
57 /// \brief Sets the maximum number of frames to cache in memory.
58 /// \param maxFrames Maximum number of frames to keep in the LRU cache.
59 /// Must be greater than 0.
60 void SetMaxCachedFrames(uint32_t maxFrames);
61
62 /// \brief Gets the maximum number of frames cached in memory.
63 /// \return The maximum number of cached frames.
64 uint32_t GetMaxCachedFrames() const;
65
66 /// \brief Initializes a new instance of the GifDecoder class.
67 GifDecoder();
68
69 /// \brief Destroys the GifDecoder and releases associated resources.
70 ~GifDecoder();
71
72 /// \brief Loads a GIF image from a file path.
73 /// \param filePath The file system path to the GIF image.
74 /// \return true if the GIF was loaded successfully; false otherwise.
75 bool LoadFromFile(const std::string& filePath);
76
77 /// \brief Loads a GIF image from an in-memory buffer.
78 /// \param data Pointer to the GIF data buffer.
79 /// \param length Length of the buffer in bytes.
80 /// \return true if the GIF was loaded successfully; false otherwise.
81 bool LoadFromMemory(const uint8_t* data, size_t length);
82
83 /// \brief Loads a GIF image from a URL.
84 /// \param url The URL to the GIF image.
85 /// \return true if the GIF was loaded successfully; false otherwise.
86 bool LoadFromUrl(const std::string& url);
87
88 /// \brief Gets the total number of frames in the GIF.
89 /// \return The number of frames, or 0 if no GIF is loaded.
90 uint32_t GetFrameCount() const;
91
92 /// \brief Gets the frame data at the specified index.
93 /// \param index The zero-based index of the frame.
94 /// \return A reference to the GifFrame at the specified index.
95 /// \throws std::out_of_range if index >= GetFrameCount().
96 const GifFrame& GetFrame(uint32_t index) const;
97
98 /// \brief Gets the width of the GIF image.
99 /// \return The width in pixels, or 0 if no GIF is loaded.
100 uint32_t GetWidth() const;
101
102 /// \brief Gets the height of the GIF image.
103 /// \return The height in pixels, or 0 if no GIF is loaded.
104 uint32_t GetHeight() const;
105
106 /// \brief Determines whether the GIF loops indefinitely.
107 /// \return true if the GIF should loop; false otherwise.
108 bool IsLooping() const;
109
110 /// \brief Gets the background color of the GIF.
111 /// \return The background color as RGBA32 (0xAABBGGRR).
112 uint32_t GetBackgroundColor() const;
113
114 /// \brief Gets BGRA pixel data with premultiplied alpha for the specified frame.
115 /// \param index The zero-based index of the frame.
116 /// \return A pointer to BGRA32 premultiplied pixel data, or nullptr on error.
117 /// The data is cached internally and valid until the next call to this function.
118 const uint8_t* GetFramePixelsBGRA32Premultiplied(uint32_t index);
119
120 /// \brief Gets BGRA pixel data with premultiplied alpha for the specified frame, scaled to
121 /// target dimensions.
122 /// \param index The zero-based index of the frame.
123 /// \param targetWidth The desired output width in pixels.
124 /// \param targetHeight The desired output height in pixels.
125 /// \param outWidth Output parameter receiving the actual output width.
126 /// \param outHeight Output parameter receiving the actual output height.
127 /// \param filter The scaling filter to use (Nearest, Bilinear, Bicubic, Lanczos).
128 /// \return A pointer to BGRA32 premultiplied scaled pixel data, or nullptr on error.
129 /// The data is cached internally and valid until the next call to this function.
130 const uint8_t* GetFramePixelsBGRA32PremultipliedScaled(
131 uint32_t index, uint32_t targetWidth, uint32_t targetHeight, uint32_t& outWidth,
132 uint32_t& outHeight, ScalingFilter filter = ScalingFilter::Bilinear);
133
134 /// \brief Starts background prefetching of frames ahead of the current playback position.
135 /// \param startFrame The frame to start prefetching from.
136 /// \remarks This starts a background thread that decodes frames ahead of playback,
137 /// reducing latency for sequential frame access.
138 void StartPrefetching(uint32_t startFrame);
139
140 /// \brief Stops background prefetching and joins the prefetch thread.
141 void StopPrefetching();
142
143 /// \brief Updates the current playback position for prefetch lookahead.
144 /// \param currentFrame The current frame being displayed.
145 /// \remarks The prefetch thread uses this to determine which frames to decode next.
146 void SetCurrentFrame(uint32_t currentFrame);
147
148 /// \brief Resets the canvas composition state for looping.
149 /// \remarks Call this when looping back to frame 0 to ensure proper frame composition.
150 /// This clears the canvas and resets disposal state without reloading the GIF.
151 void ResetCanvas();
152
153 private:
154 class Impl;
155 std::unique_ptr<Impl> _pImpl; ///< Opaque implementation (Pimpl pattern)
156};
