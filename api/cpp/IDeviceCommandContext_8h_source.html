26/// \class IDeviceCommandContext
27/// \brief Abstract interface for rendering device commands and GPU resource management.
28///
29/// Inspired by 0 A.D.'s architecture, provides backend-agnostic rendering abstraction.
30/// Enables pluggable rendering backends (DirectX 11, dummy, etc.) for maximum flexibility.
31class IDeviceCommandContext
32{
33 public:

37 /// \brief Creates a texture with the specified properties.
38 /// \param width The texture width in pixels.
39 /// \param height The texture height in pixels.
40 /// \param data Pointer to pixel data (RGBA32 format), or nullptr for empty texture.
41 /// \param dataSize Size of the pixel data in bytes.
42 /// \return A shared pointer to the created texture.

47 /// \brief Marks the beginning of a frame.

51 /// \brief Marks the end of a frame.
52 /// Finalizes all pending rendering operations.
53 virtual void EndFrame() = 0;

56 /// \param r Red channel (0.0 - 1.0).
57 /// \param g Green channel (0.0 - 1.0).

59 /// \param a Alpha channel (0.0 - 1.0).
60 virtual void Clear(float r, float g, float b, float a) = 0;

62 /// \brief Draws a texture at the specified position and size.

67 /// \param height The height to draw the texture.
68 virtual void DrawTexture(ITexture* texture, int x, int y, int width, int height) = 0;

70 /// \brief Flushes all pending rendering commands to the GPU.
71 virtual void Flush() = 0;

73 /// \brief Gets the current rendering backend type.

75 virtual Backend GetBackend() const = 0;

87 (void)outputBGRA;
88 (void)pixelCount;

92 /// \brief Scales an image using GPU acceleration with the specified filter.
93 /// \param inputBGRA Pointer to input BGRA32 premultiplied pixel data.
94 /// \param inputWidth Width of the input image.
95 /// \param inputHeight Height of the input image.
96 /// \param outputBGRA Pointer to output buffer for scaled BGRA32 pixels.
97 /// \param outputWidth Desired output width.
98 /// \param outputHeight Desired output height.
99 /// \param filter The scaling filter to use (Nearest, Bilinear, Bicubic, Lanczos).
100 /// \return true if GPU scaling succeeded; false if not supported or failed.
101 /// \note Falls back to CPU scaling if GPU acceleration is not available.
102 virtual bool ScaleImageGPU(const void* inputBGRA, uint32_t inputWidth, uint32_t inputHeight,

107 (void)inputWidth;
108 (void)inputHeight;
109 (void)outputBGRA;

111 (void)outputHeight;
