33/// \def GIFBOLT_VERSION_INT
34/// \brief Packed integer version (major * 10000 + minor * 100 + patch).
35/// Useful for compile-time version checks.
36#define GIFBOLT_VERSION_INT \
37 (GIFBOLT_VERSION_MAJOR * 10000 + GIFBOLT_VERSION_MINOR * 100 + GIFBOLT_VERSION_PATCH)

39/// \def GIFBOLT_VERSION_CHECK
40/// \brief Compile-time version comparison macro.
41/// \param major Major version to check against.
42/// \param minor Minor version to check against.
43/// \param patch Patch version to check against.
44/// \return 1 if current version >= specified version; 0 otherwise.
45#define GIFBOLT_VERSION_CHECK(major, minor, patch) \
46 (GIFBOLT_VERSION_INT >= ((major) * 10000 + (minor) * 100 + (patch)))
